def sinusoid(t, a, f, fi, d):
    grad=np.pi/180
    Y = a * np.sin((t*2*np.pi + fi*grad) * f) + d
    return Y

def pulse(t, a, ts, dur):
    t=np.array(t)
    Y = ((t >= ts) & (t < (ts + dur)))*a
    return Y
